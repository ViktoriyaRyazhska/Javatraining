public class Employee {
	 interface EmployeePay {
		int calculatePay();
	}

	private String employeeid;

	public Employee(String employeeid) {
		super();
		this.setEmployeeid(employeeid);
	}

	public Employee() {
		
	}

	@Override
	public String toString() {
		return employeeid; 
	}

	public String getEmployeeid() {
		return employeeid;
	}

	public void setEmployeeid(String employeeid) {
		this.employeeid = employeeid;
	}

	
}

public class SalariedEmployee extends Employee implements Employee.EmployeePay {
	private String name;
	private String socialSecurityNumber;
	private int rate;
	private int hours;
	
	
    @Override
	public String toString() {
		return name + " " + socialSecurityNumber + " " +  rate + " " + hours;
				
	}


	public SalariedEmployee(String employeeid, String name, String socialSecurityNumber, int rate, int hours) {
		super();
		this.socialSecurityNumber = socialSecurityNumber;
		this.rate = rate;
		this.hours = hours;
		this.name = name;
	}
	


	public int calculatePay() {
		
		return hours * rate;
	}

}

public class ContractEmployee extends Employee implements Employee.EmployeePay {
	private String name;
	private String federalTaxIdmember;
	private int monthly_payment;


	public ContractEmployee(String employeeid, String name, String federalTaxIdmember, int monthly_payment) {
		super(employeeid);
		this.federalTaxIdmember = federalTaxIdmember;
		this.monthly_payment = monthly_payment;
		this.name = name;
	}

	@Override
	public String toString() {
		return name + " " + federalTaxIdmember + " " +  monthly_payment;
				
	}

	@Override
	public int calculatePay() {
		return monthly_payment;
		
	}

}
